# Installation Instructions
# =============================================================================
#
# Install all test dependencies:
#   pip install -r requirements-test.txt
#
# Install for development (includes test deps):
#   pip install -r requirements.txt -r requirements-test.txt
#
# Run tests:
#   pytest                               # Run all tests
#   pytest -m security                   # Run security tests only
#   pytest -m "security and authentication"  # Run specific category
#   pytest --cov=app --cov-report=html   # Run with coverage
#   pytest -n auto                       # Run in parallel
#
# Security scanning:
#   safety check                         # Scan for vulnerabilities
#   bandit -r app/                       # SAST scan
#   semgrep --config=auto app/           # Advanced SAST
#
# Mutation testing:
#   mutmut run                           # Run mutation tests
#
# Performance testing:
#   locust -f tests/performance/locustfile.py  # Load testing
